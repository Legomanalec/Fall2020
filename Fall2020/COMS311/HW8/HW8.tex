\begin{lstlisting}

for(i = 2; i < M; i++)
    R(i, 1) = R(i - 1,1) + DiamondValue(i,1) - 2
    
//right
for(j = 2; i < N; i++)
    R(1, j) = R(1,j - 1) + DiamondValue(1,j) - 2
    
//diagonal
for(i = 2; i < M; i++)
    for(j = 2; i < N; i++)
        R(i, j) = 
            max(R(i - 1,1), R(1,j - 1), R(i - 1, j - 1) + DiamondValue(i, j)

return R(M, N)

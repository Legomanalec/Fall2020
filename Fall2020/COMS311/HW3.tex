{
	pop u from the head of Q;

	{
		if(w.explored == false)
		{
			add w to the tail of Q;
			w.explored = true;
			w.layer = u.layer+1;
			if(w.layer is even)
				add w to group a;
			else
				add w to group b
				
			for all neighbors y of node w
			{
				if(y.layer % 2 == w.layer % 2)
					return false;
			}
		}
	}
}
